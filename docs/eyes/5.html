    angle = math.atan2(distance_y, distance_x)

    pupil_y = eye_y + (math.sin(angle) * distance)

    screen.draw.filled_circle((pupil_x, pupil_y), 15, color=(0, 0, 100))

    screen.draw.text(
        'distance x: ' + str(distance_x)
